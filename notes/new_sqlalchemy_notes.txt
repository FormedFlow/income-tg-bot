UPDATE
Может использоваться с ORM классами и с Table, в самом простом случае, обновление одной таблицы выглядит как
from sqlalchemy import update
stmt = (
    update(user_table)
    .where(user_table.c.name == "patrick")
    .values(fullname="Patrick the Star")
)
print(stmt)


Для обновления нескольких строк в стиле executemany нужно использовать функцию sqlalchemy.bindparam()
from sqlalchemy import bindparam
stmt = (
    update(user_table)
    .where(user_table.c.name == bindparam("oldname"))
    .values(name=bindparam("newname"))
)
with engine.begin() as conn:
    conn.execute(
        stmt,
        [
            {"oldname": "jack", "newname": "ed"},
            {"oldname": "wendy", "newname": "mary"},
            {"oldname": "jim", "newname": "jake"},
        ],
    )

С её помощью помечаются так скажем плэйсхолдеры куда подставляются значения.




SQLAchemy позволяет создавать поддерживаемые многими бд бэкэндами запросы имеющие синтаксис UPDATE .. FROM:
update_stmt = (
    update(user_table)
    .where(user_table.c.id == address_table.c.user_id)
    .where(address_table.c.email_address == "patrick@aol.com")
    .values(fullname="Pat")
)
print(update_stmt)

UPDATE user_account SET fullname=:fullname FROM address
WHERE user_account.id = address.user_id AND address.email_address = :email_address_1



Помимо executemany-style запроса с использованием bindparam() можно также использовать класс sqlalchemy.Values.
Я так понял это просто контейнер, для которого сразу указываются разные параметры, а потом они используются в .where() и .values()
from sqlalchemy import Values
values = Values(
    user_table.c.id,
    user_table.c.name,
    name="my_values",
).data([(1, "new_name"), (2, "another_name"), ("3", "name_name")])
update_stmt = (
    user_table.update().values(name=values.c.name).where(user_table.c.id == values.c.id)
)
from sqlalchemy.dialects import postgresql
print(update_stmt.compile(dialect=postgresql.dialect()))

UPDATE user_account
SET name=my_values.name
FROM (VALUES (%(param_1)s, %(param_2)s), (%(param_3)s, %(param_4)s), (%(param_5)s, %(param_6)s)) AS my_values (id, name)
WHERE user_account.id = my_values.id



Ещё одна эксклюзивная (вроде) фича mySQL - порядок обновляемых полей в строках имеет значение. Т.е. можно обновить одно поле, и в обновлении этой же строки сослаться на уже обновленное значение.
update_stmt = update(some_table).ordered_values(
    (some_table.c.y, 20), (some_table.c.x, some_table.c.y + 10)
)
print(update_stmt)

UPDATE some_table SET y=:y, x=(some_table.y + :y_1)




DELETE
Тоже поддерживает Table и ORM классы.
Базовый синтаксис
from sqlalchemy import delete
stmt = delete(user_table).where(user_table.c.name == "patrick")
print(stmt)

Поддерживает коррелирующие подзапросы и синтаксис с использованием нескольких таблиц DELETE FROM .. USING (mySQL specific)
delete_stmt = (
    delete(user_table)
    .where(user_table.c.id == address_table.c.user_id)
    .where(address_table.c.email_address == "patrick@aol.com")
)
from sqlalchemy.dialects import mysql
print(delete_stmt.compile(dialect=mysql.dialect()))

DELETE FROM user_account USING user_account, address
WHERE user_account.id = address.user_id AND address.email_address = %s



update() и delete() при выполнении запросов через объект Connection хранят кол-во отобранных строк в CursorResult.rowcount
with engine.begin() as conn:
    result = conn.execute(
        update(user_table)
        .values(fullname="Patrick McStar")
        .where(user_table.c.name == "patrick")
    )
    print(result.rowcount)


Атрибут .rowcount может быть недоступен при использовании в запросе RETURNING или если это executemany запрос, это зависит от конкретного драйвера СУБД. В случае если не удалось определить .rowcount для запроса, этот атрибут вернёт -1. Некоторые драйвера, в частности, для нереляционных баз данных, могут вообще не поддерживать атрибут .rowcount.


update() и delete() поддерживают метод .returning(), возвращающий указанные поля всех строк, которые были отобраны условием WHERE. Поддерживается не всеми бэкэндами. Возвращает список строк с указанными полями.
update_stmt = (
    update(user_table)
    .where(user_table.c.name == "patrick")
    .values(fullname="Patrick the Star")
    .returning(user_table.c.id, user_table.c.name)
)


UPDATE user_account SET fullname=:fullname
WHERE user_account.name = :name_1
RETURNING user_account.id, user_account.name


print(update_stmt)
delete_stmt = (
    delete(user_table)
    .where(user_table.c.name == "patrick")
    .returning(user_table.c.id, user_table.c.name)
)
print(delete_stmt)


DELETE FROM user_account
WHERE user_account.name = :name_1
RETURNING user_account.id, user_account.name




Data manipulation with ORM
Для того чтобы с помощью ORM осуществить вставку записей в таблицу, необходимо сперва добавить их в Session с помощью session.add(), затем после добавления всех необходимых записей либо использовать session.flush() вручную, либо просто session.commit() который автоматически делает .flush().
Созданные, но ещё не добавленные в сессию объекты ORM классов висят с состоянии transient - не ассоциированные с сессией.
Поскольку при флашинге объектов, сессия должна знать их первичный ключ, вставка множества записей обычно разбивается на одиночные INSERT запросы вместо executemany-стиля. Похоже, что какие-то драйверы БД, как минимум psycopg2, способны обходить это ограничение, при этом давая всё тот же ожидаемый результат.



Identity map - механизм сессии, обечпечивающий одновременное существование лишь одного python-объекта для каждой строки из БД в рамках одной сессии. Позволяет избежать проблем рассинхронизации, неизбежных для случая когда у нас есть несколько разных объектов представляющих одну и ту же запись из таблицы:

obj1 = Session.get(User, 5) # некий User(name='бабинский')
obj2 = Session.get(User, 5) # снова бабинский но другой объект
obj1.name = 'nobody'
obj1.name == obj2.name # False


UPDATE
Основной способ обновления строк в рамках ORM подхода основан на автоматической генерации SQL инструкций, привязанных к первичному ключу строки, при изменении атрибутов ORM объекта.
Рассмотрим следующий пример:
sandy = session.execute(select(User).filter_by(name="sandy")).scalar_one()

sandy - прокси-объект для строки базы данных в рамках данной транзакции, имеет первичный ключ = 2.
sandy
User(id=2, name='sandy', fullname='Sandy Cheeks')

При изменении атрибута этого объекта
sandy.fullname = "Sandy Squirrel"

объект появится в коллекции .dirty сессии, обозначая что объект "грязный"
sandy in session.dirty
True


Теперь если сделать запрос
sandy_fullname = session.execute(select(User.fullname).where(User.id == 2)).scalar_one()
print(sandy_fullname)


Сперва благодаря autoflush произойдёт очистка буфера сессии (и выполнение соответствующих SQL инструкций для pending объектов)
UPDATE user_account SET fullname=? WHERE user_account.id = ?
[...] ('Sandy Squirrel', 2)
SELECT user_account.fullname
FROM user_account
WHERE user_account.id = ?
[...] (2,)


Объект sandy больше не находится в коллекции .dirty
sandy in session.dirty
False


Но на всякий случай снова отмечу, что без явного session.commit() обновленные данные не возымеют никакого эффекта на базу данных.




Bulk / Multi Row INSERT, upsert, UPDATE and DELETE
Для вставки, обновления или удаления большого кол-ва объектов средствами ORM существует специальный подход, называемый ORM-Enabled INSERT, UPDATE, DELETE, про который можно почитать отдельно в разделе ORM quierying guide.



Rollback сессии
session.rollback() откатывает текущую транзакцию и экспайрит атрибуты объектов с которыми мы работали в рамках сессии. Повторное обращение к таким атрибутам, как и после коммита, так и после роллбэка, приведёт к созданию нового SQL запроса, который получит из БД актуальные данные. Удаленные объекты так же возвращают своё состояние:
patrick in session
True



Closing a Session
Закрытие сессии:
1. Прекращает работу с объектом Connection, отменяет все незавершенные транзакции
2. Удаляет из сессии все объекты, они переходят в состояние detached, поскольку больше не имеют связи с объектом сессии.
Как говорит документация - лучше избегать взаимодействия с detached объектами, в крайнем случае, если необходимо, устанавливать для сессии флаг Session.expire_on_commit = False.





Working with ORM Related Objects

Тут будет в основном про relationships.
Итак, допустим мы имеем объект User
u1 = User(name='фелини', fullname='фелини бабински')

и некой Address:
a1 = Address(email_address='babinski@yahoo.com')

используя relationship, созданный на этапе построения ORM классов, как отдельный mapped атрибут, мы можем связать эти два объекта так:
u1.addresses.append(a1)


И теперь два объекта взаимосвязаны, к обоим можно получить доступ через другой
address = u1.addresses[0]
user = a1.user


Однако, пока объекты не добавлены в сессию, они висят в состоянии transient.
При добавлении в сессию "главного" объекта User (u1), будут добавлены также и связанные с ним посредством relationship объекты:
session.add(u1)
u1 in session
True
a1 in session
True


Такое поведение называется save-update cascade.



Чтобы выполнить джойн двух таблиц с помощью relationship, можно указать relationship атрибут внутри метода .join(), он будет одновременно указывать на таблицу с которой делается джойн, и в то же время из него подхватится условие ON.
print(select(Address.email_address).select_from(User).join(User.addresses))

SELECT address.email_address
FROM user_account JOIN address ON user_account.id = address.user_id


.join() и .join_from() будут использовать relationship атрибуты только в случае если они явно переданы в качестве аргумента.



В документации есть отдельный раздел посвященный "SQL generation helpers" используемых вместе с relationship, в основном для построения условия WHERE.





Loader strategies.
При загрузке из бд в память строки в виде объекта ORM класса, если у такового имелись relationship, а у этих связанных объектов так же могут быть свои relationship, очевидно, что при стандартном поведении lazy loading подгрузка таких объектов может закончиться формированием десятков отдельных запросов, что негативно скажется на производительности приложения. Поэтому в SQLAlchemy предусмотрена возможность конфигурировать поведение объектов связанных между собой с помощью relationship при загрузке.
Стоит отметить что дефолтное поведение lazy loading вообще несовместимо с asyncio.
Для того, чтобы эффективно работать со стратегими загрузки отношений, нужно покрыть обращения к БД тестами, включить echoing и смотреть за выполняемыми запросами. Если SQLAlchemy формирует слишком много SELECT запросов, вероятно может помочь смена стратегии загрузки отношений.


Loader strategies представляют из себя объекты, которые можно присоединять к SELECT запросам с помощью Select.options():
u1 = select(User).options(selectinload(User.adresses))


Либо указывать стратегию загрузки по умолчанию прямо в relationship:
from sqlalchemy.orm import Mapped
from sqlalchemy.orm import relationship


class User(Base):
    __tablename__ = "user_account"

    addresses: Mapped[List["Address"]] = relationship(
        back_populates="user", lazy="selectin"
    )


Далее доступные варианты стратегий

Select IN loading
При построении запроса в котором выбираем объекты User, и его выполнении с помощью session.execute() в действительности вместо одного запроса будет выполнено два, второй подхватит связанные объекты Address для уже загруженных User:
from sqlalchemy.orm import selectinload
stmt = select(User).options(selectinload(User.addresses)).order_by(User.id)
for row in session.execute(stmt):
    print(
        f"{row.User.name}  ({', '.join(a.email_address for a in row.User.addresses)})"
    )

FROM user_account ORDER BY user_account.id

joined load
sqlalchemy.orm.joinedload подгружает связанные отношениями объекты с помощью запросов с join.
Это в свою очередь может приводить к нежелательному дублированию строк резльтирующей выборки, в случае если у записей "главной" таблицы есть большое количество записей в "подчиненной" таблице.
from sqlalchemy.orm import joinedload
stmt = (
    select(Address)
    .options(joinedload(Address.user, innerjoin=True))
    .order_by(Address.id)
)
for row in session.execute(stmt):
    print(f"{row.Address.email_address} {row.Address.user.name}")

Raiseload
Своего рода заглушка главным образом предназначенная для отслеживания проблемы N+1, выбрасывающая исключение sqlalchemy.exc.InvalidRequestError при попытке ленивой загрузки relationship объектов или коллекций.
Используется путём указания relationship.lazy='raise_on_sql'. Вынуждает при использовании select запросов направленных на получение объектов имеющих relationship явно использовать другие стратегии их загрузки.
lazy='raise_on_sql' пытается исключать лишнюю работу там где это возможно. Скажем, если есть select() запрос выбирающий объекты Address, и встречается Address.user, ещё не загруженный с помощью SQL, но уже имеющийся в сессии, то он возьмётся из сессии без повторной загрузки.
